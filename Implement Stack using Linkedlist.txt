struct node
{
int info;
srtuct node *next;
};
typedef struct node stack;
stack * top=NULL;
void push(int);
int pop();
void display();
void push(int item)
{
stack *newnode;
newnode=(stack*)malloc(sizeof(stack))
newnode->info=item;
newnode->next=top;
top=newnode;
}
int pop()
{
stack *n;
int x;
if(top==NULL)
{
printf("Stack Underflow\n");
return 0;
}
else
{
n=top;
top=n->next;
x=n->info;
free(n);
return x;
}
}
void display()
{
stack *i;
printf("Elements of STack are\n");
for(i=top;i!=NULL;i=i->next)
{
printf("%d",i->info);
}
}
